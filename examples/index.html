var layer = new goolib.Layer("div", {
  position: "absolute",
  top: "50%",
  left: "50%",
  background: "#006262",
  color: "#fff",
  x: "-50%",
  y: "-50%",
  cursor: "pointer",
  padding: "14px"
})
layer.text("I am button");
layer.addTo(document.querySelector(".example-2"));

layer.on("mouseenter", function() {
  layer.to(.2, {
    scale: 2
  })
});
layer.on("mouseleave", function() {
  layer.to(.2, {
    scale: 1
  })
});
layer.on("click", function() {
  layer.fromTo(.2, {
    rotation: 0
  }, {
    rotation: 360
  })
});

Layers in layers


var layer = new goolib.Layer("div", {
  padding: "16px"
});
layer.addTo(document.querySelector('.example-3'));

for (var i = 0; i < 6; i++) {
  var nested = new goolib.Layer("div", {
    padding: "10px"
  });
  nested.text("Layer " + i);
  nested.on("mouseenter", function() {
    TweenMax.to(this, .2, {
      background: "#98ff67",
    })
  });
  nested.on("mouseleave", function() {
    TweenMax.to(this, .2, {
      background: "#e2ffd5"
    })
  });

  layer.addChild(nested);
}
